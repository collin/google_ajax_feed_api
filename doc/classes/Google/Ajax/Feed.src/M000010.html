      def load options={}
        url = self.class.api.load_query @url, options
        
        # Very strange json bug. Bye bye tabs        
        http_response = JSON.parse open(url).read.gsub("\t", '')
        
        @feed = http_response["responseData"]["feed"]
        
        return @feed.length
      end
