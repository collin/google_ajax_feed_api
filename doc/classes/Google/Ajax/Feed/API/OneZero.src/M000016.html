            def load_params options={} #:nodoc
              options = ({
                :limit => Feed.config.limit,
                :history => Feed.config.history
              }).merge(options)
              
              params = "&num=#{options[:limit]}"
              params << '&scoring=h' if options[:history]
              params        
            end
